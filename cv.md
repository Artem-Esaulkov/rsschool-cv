function nbDig(n, d) {
  let array = [];
  let counter = 0;
  for (let i = 0; i <= n; i++) {
    let newNumber = String(i**2);
    array.push(newNumber);
  };
  array = array.join('');
  for (let i = 0; i < array.length; i++) {
    if (array[i] === String(d)) {
      counter += 1;
    }
  }
  
  return counter;
};
